# 生成 gpg 密钥
gpg --gen-key

# 生成吊销证书
gpg --gen-revoke 0102030405060708

# 列出所有 gpg 公钥
gpg --list-keys

# 列出所有 gpg 私钥
gpg --list-secret-keys

# 删除 gpg 公钥
gpg --delete-keys 0102030405060708

# 删除 gpg 私钥
gpg --delete-secret-keys 0102030405060708

# 输出 gpg 公钥 ascii
gpg --armor --output public.key --export 0102030405060708

# 输出 gpg 私钥 ascii
gpg --armor --output private.key --export-secret-keys 0102030405060708

# 上传 gpg 公钥
gpg --send-keys 0102030405060708 --keyserver 

# 查看 gpg 公钥指纹
gpg --fingerprint 0102030405060708

# 导入 gpg 密钥(导入私钥时会自动导入公钥)
gpg --import private.key

# 加密文件
gpg --recipient 0102030405060708 --output encrypt.file --encrypt origin.file

# 解密文件
gpg --output origin.file --decrypt encrypt.file

# 文件签名，生成二进制的 gpg 文件
gpg --sign plaintext_file

# 文件签名，生成文本末尾追加 ASCII 签名的 asc 文件
gpg --clearsign plaintext_file

# 文件签名，生成二进制的 sig 文件
gpg --detach-sign plaintext_file

# 文件签名，生成 ASCII 格式的 asc 文件
gpg --detach-sign plaintext_file

# 签名并加密
gpg --local-user 0102030405060708 --recipient 0102030405060708 --armor --sign --encrypt plaintext_file

# 验证签名
gpg --verify plaintext_file.asc plaintext_file
