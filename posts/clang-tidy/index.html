// main.cc 
#include <algorithm>
#include <iostream>
#include <string>
#include <vector>

class PipeLine {
 public:
  std::vector<std::string> cmds;
};

int main(int argc, char *argv[]) {
  (void)argc;
  (void)argv;
  PipeLine pl;
  pl.cmds.push_back("ls");
  pl.cmds.push_back("grep");
  pl.cmds.push_back("*.cc");
  std::for_each(pl.cmds.begin(), pl.cmds.end(),
                [](const auto &item) { std::cout << item << std::endl; });
  return 0;
}

set_property(TARGET tidy PROPERTY CXX_STANDARD_REQUIRED ON)
set_property(TARGET tidy PROPERTY CXX_EXTENSIONS OFF)
set(CMAKE_EXPORT_COMPILE_COMMANDS "ON")
set(CMAKE_BUILD_TYPE Debug)
target_sources(tidy PRIVATE 
    main.cc
    )
