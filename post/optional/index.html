class Entity {
    public:
    explicit Entity(uint64_t id)
        : id_(id) {};
    ~Entity() = default;
    private:
    [[maybe_unused]] uint64_t id_;
};

std::map<uint64_t, Entity> entitys;

Entity find_entity(uint64_t id)
{
    auto it = entitys.find(id);
    if(it == entitys.end())
    {
        return Entity{};  // error
    }
    return it->second;
}

std::optional<Entity> find_entity(uint64_t id)
{
    auto it = entitys.find(id);
    if (it == entitys.end()) {
        return {};
    }
    return it->second;
}
