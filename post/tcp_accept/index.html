basic_socket_acceptor 类虽然方法有很多，不过因为使用 implementation 技术的原因，重点还是关注成员变量impl_

template <typename Protocol, typename Executor = any_io_executor>
class basic_socket_acceptor;
template <typename Protocol, typename Executor>
class basic_socket_acceptor
  : public socket_base
{
#if defined(ASIO_WINDOWS_RUNTIME)
  detail::io_object_impl<
    detail::null_socket_service<Protocol>, Executor> impl_;
#elif defined(ASIO_HAS_IOCP)
  detail::io_object_impl<
    detail::win_iocp_socket_service<Protocol>, Executor> impl_;
#else
  detail::io_object_impl<
    detail::reactive_socket_service<Protocol>, Executor> impl_; // Protocol 既是 tcp
#endif
};

template <typename IoObjectService,
    typename Executor = io_context::executor_type>
class io_object_impl
{
public:
  typedef IoObjectService service_type; // IoObjectService 既是 detail::reactive_socket_service<Protocol>
  typedef typename service_type::implementation_type implementation_type;
  typedef Executor executor_type; // Executor 是 basic_socket_acceptor 类模版的第二个模板类型

  explicit io_object_impl(int, const executor_type& ex)
    : service_(&asio::use_service<IoObjectService>(
          io_object_impl::get_context(ex))),
      executor_(ex)
  {
    service_->construct(implementation_);
  }
  template <typename ExecutionContext>
  explicit io_object_impl(int, int, ExecutionContext& context)
    : service_(&asio::use_service<IoObjectService>(context)),
      executor_(context.get_executor())
  {
    service_->construct(implementation_);
  }
private:
  service_type* service_;
  implementation_type implementation_;
  executor_type executor_;
};

io_object_impl 类成员变量 service_type 的真实类型是 IoObjectService
对于 basic_socket_acceptor 类成员变量 impl_ 的定义是 detail::io_object_impl< detail::reactive_socket_service<Protocol>, Executor>
对应的 io_object_impl 的模板参数 IoObjectService 就是 detail::reactive_socket_service<Protocol>

service_ 的类型就是 detail::reactive_socket_service<Protocol>
executor_ 的类型就是 basic_socket_acceptor 类模版的第二个模板类型
implementation_ 的类型是 detail::reactive_socket_service<Protocol>::implementation_type
