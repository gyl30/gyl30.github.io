io_context::impl_type& io_context::add_impl(io_context::impl_type* impl)
{
  asio::detail::scoped_ptr<impl_type> scoped_impl(impl); // 智能指针，如果下面的逻辑出现了异常，会自动释放内存

inline void add_service(execution_context& e, Service* svc)
{
  // Check that Service meets the necessary type requirements. 类型检查
  (void)static_cast<execution_context::service*>(static_cast<Service*>(0));

  e.service_registry_->template add_service<Service>(svc); // 服务添加
}

template <typename Service>
void service_registry::add_service(Service* new_service)
{
  execution_context::service::key key;
  init_key<Service>(key, 0); // 构造唯一标示
  return do_add_service(key, new_service); // 添加服务
}

//

void service_registry::do_add_service(
    const execution_context::service::key& key,
    execution_context::service* new_service)
{
  if (&owner_ != &new_service->context()) // context 不一致
    asio::detail::throw_exception(invalid_service_owner());

  asio::detail::mutex::scoped_lock lock(mutex_);

  // Check if there is an existing service object with the given key.
  execution_context::service* service = first_service_;
  while (service) // 检查是否已经存在
  {
    if (keys_match(service->key_, key))
      asio::detail::throw_exception(service_already_exists());
    service = service->next_;
  }

  // Take ownership of the service object. // 添加到服务链表中
  new_service->key_ = key;
  new_service->next_ = first_service_;
  first_service_ = new_service;
}

post 函数通过层层的模板调用，最终将请求转发至 io_context::initiate_post 这个函数对象内，通过模版的实例化在编译期间检查类型，并根据类型查找最佳调用

initiate_post 的实现如下

struct io_context::initiate_post
{
  template <typename LegacyCompletionHandler>
  void operator()(ASIO_MOVE_ARG(LegacyCompletionHandler) handler,
      io_context* self) const
  {
    // If you get an error on the following line it means that your handler does
    // not meet the documented type requirements for a LegacyCompletionHandler.
    ASIO_LEGACY_COMPLETION_HANDLER_CHECK( LegacyCompletionHandler, handler) type_check; //  类型检查

    detail::non_const_lvalue<LegacyCompletionHandler> handler2(handler);

    bool is_continuation = asio_handler_cont_helpers::is_continuation(handler2.value); // 是否延迟执行 默认是 false

    // Allocate and construct an operation to wrap the handler.
    typedef detail::completion_handler< typename decay<LegacyCompletionHandler>::type, 
    executor_type> op; // op 的真实类型是 completion_handler
    typename op::ptr p = { detail::addressof(handler2.value),
        op::ptr::allocate(handler2.value), 0 };

    p.p = new (p.v) op(handler2.value, self->get_executor());

    ASIO_HANDLER_CREATION((*self, *p.p,
          "io_context", self, 0, "post"));

    self->impl_.post_immediate_completion(p.p, is_continuation);
    p.v = p.p = 0;
  }
};

通过层层包装，将函数post传递的回调函数包装为 completion_handler 类型的指针变量,并将指针转交给了 sceduler 的 post_immediate_completion 函数
is_continuation 的默认实现为 false，可以重写覆盖默认行为

void scheduler::post_immediate_completion(
    scheduler::operation* op, bool is_continuation)
{
  if (one_thread_ || is_continuation) // 如果需要延迟执行，放入到当前线程的本地队列中
  {
    if (thread_info_base* this_thread = thread_call_stack::contains(this))
    {
      ++static_cast<thread_info*>(this_thread)->private_outstanding_work;
      static_cast<thread_info*>(this_thread)->private_op_queue.push(op);
      return;
    }
  }
  work_started();
  mutex::scoped_lock lock(mutex_); 
  op_queue_.push(op); 
  wake_one_thread_and_unlock(lock);
}

  count_type s = impl_.run(ec);  // io_context的 run 函数什么都没做，直接将调用转交给了 scheduler
  asio::detail::throw_error(ec);

// scheduler

std::size_t scheduler::run(asio::error_code& ec)
{
  ec = asio::error_code();
  if (outstanding_work_ == 0)
  {
    stop();
    return 0;
  }
    
  // 我们知道 run 函数返回就代表结束事件循环了，正常情况下是不会结束的，
  // 所以在循环之前先构建一个当前线程的上下文，之后将线程上下文注册到  thread_call_stack 上

  thread_info this_thread; 
  this_thread.private_outstanding_work = 0;
  thread_call_stack::context ctx(this, this_thread);

  mutex::scoped_lock lock(mutex_);

  std::size_t n = 0;
  for (; do_run_one(lock, this_thread, ec); lock.lock()) // 开始事件循环
    if (n != (std::numeric_limits<std::size_t>::max)())
      ++n;
  return n;
}


// do_run_one

std::size_t scheduler::do_run_one(mutex::scoped_lock& lock,
    scheduler::thread_info& this_thread,
    const asio::error_code& ec)
{
  while (!stopped_)
  {
    if (!op_queue_.empty()) // 检查完成队列中有没有需要调用的对象
    {
      // Prepare to execute first handler from queue.
      operation* o = op_queue_.front();
      op_queue_.pop();
      bool more_handlers = (!op_queue_.empty());

      if (o == &task_operation_) // task_operation_ 是一个特殊的哨兵对象
      {
        task_interrupted_ = more_handlers; // 需不需要立即触发事件循环
        // 如果还有更多的回调对象需要调用,并且不止一个线程,通知其他线程干活
        if (more_handlers && !one_thread_)
          wakeup_event_.unlock_and_signal_one(lock);
        else
          lock.unlock();

        task_cleanup on_exit = { this, &lock, &this_thread }; 
        (void)on_exit;

        // Run the task. May throw an exception. 
        Only block if the operation queue is empty and we're not polling, 
        otherwise we want to return as soon as possible.
        task_->run(more_handlers ? 0 : -1, this_thread.private_op_queue); 
        // task_->run 就是 epoll_wait
        // epoll_wait 返回的就绪事件放在 this_thread.private_op_queue 中
        // task_cleanup 的析构函数会将 this_thread.private_op_queue 中的就绪任务放到 scheduler 的 op_queue_ 中
      }
      else
      {
        // 从 op_queue_ 中拿到的任务不是哨兵，就是正常的完成事件的回调函数

        std::size_t task_result = o->task_result_; 

        if (more_handlers && !one_thread_) // 唤醒其他线程继续处理剩下的任务
          wake_one_thread_and_unlock(lock);
        else
          lock.unlock();
        // Ensure the count of outstanding work is decremented on block exit.
        work_cleanup on_exit = { this, &lock, &this_thread };
        (void)on_exit;

        // Complete the operation. May throw an exception. Deletes the object.
        o->complete(this, ec, task_result); // 调用用户的回调函数
        this_thread.rethrow_pending_exception();

        return 1;
      }
    }
    else
    {
      wakeup_event_.clear(lock);
      wakeup_event_.wait(lock);
    }
  }

  return 0;
}
